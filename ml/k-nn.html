import numpy as np
import pandas as panda
import matplotlib.pyplot as plt

from sklearn.datasets import fetch_mldata
from sklearn.model_selection import train_test_split
from sklearn import neighbors

# BEGINNING


print("--- Mise en pratique de l'algo k-NN ---\n")


# DATA LOADING


print("Chargement des donnes depuis fichier local...\n")


mnist = fetch_mldata('MNIST original', data_home="./")


print("Nombre d'images avec leur chiffre rsolu = " + str(mnist.target.shape[0]) + "\n")


# SAMPLING


print("Echantillonnage des donnes -> 5000 parmi 70000 ...\n")

# best pratice = sklearn.utils.resample pour eviter les doublons
sample = np.random.randint(70000, size=5000)

sampledData = mnist.data[sample]
sampledTarget = mnist.target[sample]

print("Splitting des donnes (80% pour le training / 20% pour le test) ...\n")

xtrain, xtest, ytrain, ytest = train_test_split(sampledData, sampledTarget, train_size=0.8, test_size=0.2)

# LEARNING


print("Apprentissage avec l'algo k-NN (k=3) sur les donnes de training...\n")

knn = neighbors.KNeighborsClassifier(n_neighbors=3)
knn.fit(xtrain, ytrain)


# CALCUL DE L'ERREUR

error = (1 - knn.score(xtest, ytest)) * 100

print('Pourcentage d\'erreur sur donnes de test (k = 3) = %f\n' % error)

print('Entrainement termin, prdiction et affichage de 12 images au hasard\n(sur donnes de test - non entraines)\n')

# On rcupre les prdictions sur les donnes test
predicted = knn.predict(xtest)

# On redimensionne les donnes sous forme d'images
images = xtest.reshape((-1, 28, 28))

# On selectionne un echantillon de 12 images au hasard
select = np.random.randint(images.shape[0], size=12)

# On affiche les images avec la prdiction associe
for index, value in enumerate(select):
    plt.subplot(3,4,index+1)
    plt.axis('off')
    plt.imshow(images[value],cmap=plt.cm.gray_r,interpolation="nearest")
    plt.title('Prdiction: %i' % predicted[value])

plt.show()

# OPTIMIZATION DU MODELE

print('Entrainement sur donnes de test (de k = 2  7)...\n')

errors = []
for k in range(2,7):
    knn = neighbors.KNeighborsClassifier(k)
    errors.append(100*(1 - knn.fit(xtrain, ytrain).score(xtest, ytest)))

print('Pourcentage d\'erreurs de prdiction (pour k = 2  7) :\n')    
    
plt.plot(range(2,7), errors, 'o-')
plt.show()

print('Exemple de 12 images mal prdites (pour k = 3) :\n')

# on rcupre les donnes mal prdites 

misclass = (ytest != predicted)
misclass_images = images[misclass,:,:]
misclass_predicted = predicted[misclass]

# on slectionne un chantillon de ces images
select = np.random.randint(misclass_images.shape[0], size=12)

# on affiche les images et les prdictions (errones) associes  ces images
for index, value in enumerate(select):
    plt.subplot(3,4,index+1)
    plt.axis('off')
    plt.imshow(misclass_images[value],cmap=plt.cm.gray_r,interpolation="nearest")
    plt.title('Predicted: %i' % misclass_predicted[value])

plt.show()
